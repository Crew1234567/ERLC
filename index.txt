  <script>
    function sortTable(columnIndex) {
      const table = document.getElementById("vehicleTable");
      const rows = Array.from(table.rows).slice(1);
      const sorted = rows.sort((a, b) => {
        const aText = a.cells[columnIndex].innerText;
        const bText = b.cells[columnIndex].innerText;
        const aVal = columnIndex === 2 ? parseFloat(aText.replace(/[^0-9.]/g, "")) : aText.toLowerCase();
        const bVal = columnIndex === 2 ? parseFloat(bText.replace(/[^0-9.]/g, "")) : bText.toLowerCase();
        return aVal > bVal ? 1 : -1;
      });
      sorted.forEach(row => table.tBodies[0].appendChild(row));
    }

    document.getElementById("searchInput").addEventListener("input", function () {
      const filter = this.value.toLowerCase();
      const rows = document.querySelectorAll("#vehicleTable tbody tr");
      rows.forEach(row => {
        const name = row.cells[1].innerText.toLowerCase();
        row.style.display = name.includes(filter) ? "" : "none";
      });
    });

    document.getElementById("typeFilter").addEventListener("change", function () {
      const filter = this.value;
      const rows = document.querySelectorAll("#vehicleTable tbody tr");
      rows.forEach(row => {
        const type = row.cells[3].innerText;
        row.style.display = !filter || type === filter ? "" : "none";
      });
    });
  </script>
